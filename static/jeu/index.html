// URLs des images pour chaque face du dé const imageUrls = ['https://www.calculator.net/img/dice1.png', 'https://www.calculator.net/img/dice2.png', 'https://www.calculator.net/img/dice3.png', 'https://www.calculator.net/img/dice4.png', 'https://www.calculator.net/img/dice5.png', 'https://www.calculator.net/img/dice6.png', ]; // Charger les textures pour chaque face du dé const textures = imageUrls.map(url => new THREE.TextureLoader().load(url)); // Créer un matériau avec les textures chargées const materialArray = textures.map(texture => new THREE.MeshBasicMaterial({ map: texture })); // Créer un matériau composé des textures const material = new THREE.MeshFaceMaterial(materialArray); // Création d'un dé avec 6 faces const createDice = () => { const geometry = new THREE.BoxGeometry(); return new THREE.Mesh(geometry, material); }; // Ajout de 8 dés à la scène const dice = Array.from({ length: 8 }, createDice); dice.forEach((die, index) => { die.position.x = (index % 4) * 2; die.position.y = Math.floor(index / 4) * 2; scene.add(die); }); // Fonction de lancer de dé const rollDice = () => { dice.forEach(die => { die.rotation.x += Math.random() * 2 * Math.PI; die.rotation.y += Math.random() * 2 * Math.PI; die.rotation.z += Math.random() * 2 * Math.PI; }); }; // Animation de la scène const animate = () => { requestAnimationFrame(animate); renderer.render(scene, camera); };
